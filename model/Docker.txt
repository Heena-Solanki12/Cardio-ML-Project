# 1️⃣ Use an official Python runtime as a base image
FROM python:3.9-slim

# 2️⃣ Set a working directory inside the container
WORKDIR /app

# 3️⃣ Copy requirements file and install dependencies
COPY requirements.txt .

RUN pip install --no-cache-dir -r requirements.txt

# 4️⃣ Copy the rest of your project files
COPY . .

# 5️⃣ (Optional) If you need to run training before serving,
#     uncomment the next line:
# RUN python train.py

# 6️⃣ Expose port if your app serves a web service:
#     (adjust port if app uses a different one)
EXPOSE 8501

# 7️⃣ Command to run your app
#    → Adjust this if your app uses Flask, Streamlit, FastAPI, etc.
CMD ["streamlit", "run", "app.py", "--server.port=8501", "--server.address=0.0.0.0"]
